library(Lahman)
library(sqldf)
library(ggplot2)

Bar Plots are simply a visualization of data using bars. Below is a Bar Plot displaying each MLB teamâ€™s homerun totals in 1980.

query<-"SELECT name,HR
FROM Teams
WHERE yearID=1980
ORDER BY HR"

result<-sqldf(query)

result$name<-factor(result$name,levels=result$name)

ggplot()+
  geom_bar(data=result,aes(x=name,y=HR),stat='identity',color="white",fill="blue")+
  coord_flip()+
  ylab("Homeruns")+
  xlab("Teams")+
  ggtitle("1980 Team Homerun Distribution")
